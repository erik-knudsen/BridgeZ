319void CBidDialog::enableBidder(Seat seat, Bids lastBid, Bids doubleBid)
320{
321 int i;
322
323 //Set window title.
324 setWindowTitle(tr("Bidder: ") + QCoreApplication::translate("defines", SEAT_NAMES[seat]));
325
326 //Enable proper bid buttons.
327 for (i = BID_1C; i <= BID_7NT; i++)
328 m_pBidButtons[i]->setEnabled(i > lastBid);
329
330 //Enable proper double/redouble buttons.
331 if (lastBid == BID_NONE)
332 doubleBid = BID_NONE;
333
334 m_pBidButtons[BID_DOUBLE]->setEnabled(doubleBid == BID_DOUBLE);
335 m_pBidButtons[BID_REDOUBLE]->setEnabled(doubleBid == BID_REDOUBLE);
336
337 //Enable the dialog.
338 setEnabled(true);
339}

352void CBidDialog::setBidHint(Bids bid)
353{
354 assert((bid >= BID_NONE) && (bid <= BID_REDOUBLE));
355
356 if (bidHint != BID_NONE)
357 m_pBidButtons[bidHint]->setGraphicsEffect(0);
358
359 if (bid != BID_NONE)
360 {
361 QGraphicsColorizeEffect *effect = new QGraphicsColorizeEffect();
362 effect->setColor(QColor(0, 0, 255));
363 m_pBidButtons[bid]->setGraphicsEffect(effect);
364 }
365
366 bidHint = bid;
367}
