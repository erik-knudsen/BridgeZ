75void CActorLocalAuto::clientActions()
76{
77 //React to client out events.
78 if (zBridgeClientIface_israised_connect(&handle))
79 {
80 //Connect to the server (upon entry of the statechart).
81 emit sConnect(teamName , (Seat)zBridgeClientIface_get_client(&handle), protocol);
82 }
83
84 else if (zBridgeClientIface_israised_rTNames(&handle))
85 {
86 //Actor is seated and ready for team names.
87 emit sRTNames( (Seat)zBridgeClientIface_get_client(&handle));
88 }
89
90 else if (zBridgeClientIface_israised_rSBoard(&handle))
91 {
92 //Actor has received team names and is ready for start of board.
93 emit sRSBoard( (Seat)zBridgeClientIface_get_client(&handle));
94 }
95
96 else if (zBridgeClientIface_israised_rDealInfo(&handle))
97 {
98 //Actor has received start of board and is ready for deal information.
99 emit sRDealInfo( (Seat)zBridgeClientIface_get_client(&handle));
100 }
101
102 else if (zBridgeClientIface_israised_rCards(&handle))
103 {
104 //Actor has received deal information and is ready for cards.
105 emit sRCards((Seat)zBridgeClientIface_get_client(&handle));
106
107 updateCurrentGameInfo = updateGameInfo;
108 }
109
110 else if (zBridgeClientIface_israised_bidInfo(&handle))
111 {
112 //Bidding is finished.
113 //Set in bid history,
114 bidAndPlay.setBidInfo((Bids)zBridgeClientIface_get_lastBid(&handle),
115 (Bids)zBridgeClientIface_get_bidDouble(&handle),
116 (Seat)zBridgeClientIface_get_bidInfo_value(&handle));
117 }
118
119 else if (zBridgeClientIface_israised_clientPlays(&handle))
120 {
121 //Actor plays a card.
122 //Must get card to play from automatic play.
123 playValue();
124 }
125
126 else if (zBridgeClientIface_israised_readyForDummyCards(&handle))
127 {
128 //Actor is ready for dummys cards.
129 emit sReadyForDummyCards((Seat)zBridgeClientIface_get_client(&handle));
130 }
131
132 else if (zBridgeClientIface_israised_getLeader(&handle))
133 {
134 //Get next leader.
135 zBridgeClientIface_raise_newLeader(&handle, bidAndPlay.getNextLeader());
136
137 //State chart run cycle.
138 clientRunCycle();
139 }
140
141 else if (zBridgeClientIface_israised_undoPlay(&handle) || zBridgeClientIface_israised_undoBid(&handle))
142 {
143 }
144
145 else if (zBridgeClientIface_israised_undoTrick(&handle))
146 {
147 }
148
149 //Can come together with undoBid and must be processed after undoBid.
150 if (zBridgeClientIface_israised_giveBid(&handle))
151 {
152 //Must get bid from automatic player.
153 //Calculate automatic bid.
154 bidValue();
155 }
156
157 //Can come together with undoBid and must be processed after undoBid.
158 if (zBridgeClientIface_israised_rBid(&handle))
159 {
160 //Actor is ready for bid.
161 emit sRBid((Seat)zBridgeClientIface_get_client(&handle), (Seat)zBridgeClientIface_get_bidder(&handle));
162 }
163
164 //Initially comes together with bidInfo and must be processed after bidInfo.
165 if (zBridgeClientIface_israised_readyForDummy(&handle))
166 {
167 //Actor is ready for dummy to play a card.
168 emit sReadyForDummy((Seat)zBridgeClientIface_get_client(&handle), zBridgeClientIface_get_noTrick(&handle));
169 }
170
171 //Initially comes together with bidInfo and must be processed after bidInfo.
172 if (zBridgeClientIface_israised_readyForPlayer(&handle))
173 {
174 //Actor is ready for player to play a card.
175 emit sReadyForPlayer((Seat)zBridgeClientIface_get_client(&handle),
176 (Seat)zBridgeClientIface_get_player(&handle), zBridgeClientIface_get_noTrick(&handle));
177 }
178
179 if (zBridgeClientIface_israised_synchronize(&handle))
180 {
181 //Synchronization of server and clients.
182 zBridgeClientSync_init(&syncHandle);
183 int syncState = zBridgeClientIface_get_syncState(&handle);
184 zBridgeClientSyncIface_set_syncState(&syncHandle, syncState);
185 synchronizing = true;
186 zBridgeClientSync_enter(&syncHandle);
187 clientSyncActions();
188 }
189}

191void CActorLocalAuto::clientSyncActions()
192{
193 //React to sync client out events.
194 if (zBridgeClientSyncIface_israised_sendAttemptSync(&syncHandle))
195 {
196 //Update Table Manager game info.
197 int syncState = zBridgeClientSyncIface_get_syncState(&syncHandle);
198 if (updateCurrentGameInfo && (syncState == SS))
199 {
200 emit sUpdateGame();
201 updateCurrentGameInfo = false;
202 }
203
204 Seat seat = (Seat)zBridgeClientIface_get_client(&handle);
205 emit sAttemptSyncFromClientToServer(seat);
206 }
207
208 else if (zBridgeClientSyncIface_israised_sendConfirmSync(&syncHandle))
209 {
210 int syncState = zBridgeClientSyncIface_get_syncState(&syncHandle);
211 //Synchronization after bid and before play?
212 if (syncState == SP)
213 {
214 Seat declarer = bidAndPlay.getBidHistory().getDeclarer();
215 Seat dummy = (Seat)((declarer + 2) % 4);
216 Seat leader = (Seat)((declarer + 1) % 4);
217 zBridgeClientIface_set_declarer(&handle, declarer);
218 zBridgeClientIface_set_dummy(&handle, dummy);
219 zBridgeClientIface_set_leader(&handle, leader);
220 zBridgeClientIface_set_player(&handle, leader);
221 }
222 //Continue.
223 continueSync();
224 }
225
226 else if (zBridgeClientSyncIface_israised_okSync(&syncHandle))
227 {
228 synchronizing = false;
229 zBridgeClientIface_raise_allSync(&handle);
230 clientRunCycle();
231 }
232}

464void CActorLocalAuto::undoBid(bool reBid)
465{
466 int undo;
467
468 if (reBid)
469 {
470 bidAndPlay.resetPlayHistory(); //Could come from play part.
471 bidAndPlay.resetBidHistory();
472 undo = REBID;
473 }
474 else
475 {
476 Bids bid;
477 undo = bidAndPlay.bidUndo(&bid);
478 if (undo >= 0)
479 zBridgeClientIface_set_bidVal(&handle, bid);
480 }
481 zBridgeClientIface_raise_undo(&handle, undo);
482 clientRunCycle();
483}

489void CActorLocalAuto::undoTrick(bool rePlay)
490{
491 int leader = NO_SEAT;
492
493 if (rePlay)
494 {
495 bidAndPlay.resetPlayHistory(false);
496 leader = REPLAY;
497 }
498 else if (zBridgeClient_isStateActive(&handle, ZBridgeClient_main_region_Play))
499 leader = bidAndPlay.playUndo(PT);
500 else if (zBridgeClient_isStateActive(&handle, ZBridgeClient_main_region_SyncLeader))
501 leader = bidAndPlay.playUndo(CT);
502
503 zBridgeClientIface_raise_undo(&handle, leader);
504 clientRunCycle();
