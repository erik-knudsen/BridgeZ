249void CActorRemoteAuto::receiveLine(QString line, bool connected)
250{
251 try
252 {
253 //Determine type.
254 MsgType msgType = ::getMessageType(line);
255 switch (msgType)
256 {
257 case CONNECT_MSG:
258 {
259 //A connect message has been received
260 CConnectMsg connectMsg(line);
261
262 //If already connected then the connect message should be discarded.
263 //The message will be received when a new session is requested from the server. In this
264 //case the client is not restarted completely from scratch, but the already established
265 //connection is kept. So the connect message is not needed.
266 if (!connected)
267 emit sConnect(connectMsg.name, connectMsg.seat, connectMsg.protocol);
268 break;
269 }
270
271 case RTNAMES_MSG:
272 {
273 //A redy for team names message has been received.
274 CRTNamesMsg rTNames(line);
275 emit sRTNames(rTNames.seat);
276 break;
277 }
278
279 case RSBOARD_MSG:
280 {
281 //A ready for start of board message has been received.
282 CRSBoardMsg rSBoard(line);
283 emit sRSBoard(rSBoard.seat);
284 break;
285 }
286
287 case RDEALINFO_MSG:
288 {
289 //A ready for deal info message has been received.
290 CRDealInfoMsg rDealInfoMsg(line);
291 emit sRDealInfo(rDealInfoMsg.seat);
292 break;
293 }
294
295 case RCARDS_MSG:
296 {
297 //A ready for cards message has been received.
298 CRCardsMsg rCardsMsg(line);
299 emit sRCards(rCardsMsg.seat);
300 break;
301 }
302
303 case BID_MSG:
304 {
305 //A bid message has been received.
306 CBidMsg bidMsg(line);
307 emit sBid(bidMsg.bidder, bidMsg.bid);
308 break;
309 }
310
311 case RBID_MSG:
312 {
313 //A ready for bid message has been received.
314 CRBidMsg rBidMsg(line);
315 emit sRBid(rBidMsg.player, rBidMsg.bidder);
316 break;
317 }
318
319 case PLAYERPLAYS_MSG:
320 {
321 //A player plays message has been received.
322 CPlayerPlaysMsg playerPlaysMsg(line);
323 emit sPlayerPlays(playerPlaysMsg.player, playerPlaysMsg.card);
324 break;
325 }
326
327 case READYFORPLAYER_MSG:
328 {
329 //A ready for player message has been received.
330 CReadyForPlayerMsg readyForPlayerMsg(line);
331 emit sReadyForPlayer(readyForPlayerMsg.seat, readyForPlayerMsg.player, readyForPlayerMsg.trick);
332 break;
333 }
334
335 case READYFORDUMMY_MSG:
336 {
337 //A ready for dummy message has been received.
338 CReadyForDummyMsg readyForDummyMsg(line);
339 emit sReadyForDummy(readyForDummyMsg.seat, readyForDummyMsg.trick);
340 break;
341 }
342
343 case READYFORDUMMYCARDS_MSG:
344 {
345 //A ready for dummys cards message has been received.
346 CReadyForDummyCardsMsg readyForDummyCardsMsg(line);
347 emit sReadyForDummyCards(readyForDummyCardsMsg.seat);
348 break;
349 }
350
351 case ATTEMPT_SYNCHRONIZE_MSG:
352 {
353 //An attempt synchronize message has been received.
354 CAttemptSynchronizeMsg attemptSynchronize(line);
355 emit sAttemptSyncFromClientToServer(attemptSynchronize.seat);
356 break;
357 }
358
359 case CONFIRM_SYNCHRONIZE_MSG:
360 {
361 //A confirm synchronize message has been received.
362 CConfirmSynchronizeMsg confirmSynchronize(line);
363 emit sConfirmSyncFromClientToServer(confirmSynchronize.seat);
364 break;
365 }
366
367 default:
368 throw NetProtocolException("Net - Illegal message: " + line.toStdString());
369 break;
370
371 }
372 }
373 catch (NetProtocolException &e)
374 {
375 //There was an error in the message.
376 QMessageBox::critical(0, tr("ZBridge"), e.what());
377 }
378}
