109void CActorLocal::clientActions()
110{
111 //React to client out events.
112 if (zBridgeClientIface_israised_connect(&handle))
113 {
114 //Connect to the server (upon entry of the statechart).
115 emit sConnect(teamName , (Seat)zBridgeClientIface_get_client(&handle), protocol);
116 }
117
118 else if (zBridgeClientIface_israised_rTNames(&handle))
119 {
120 //Actor is seated and ready for team names.
121 emit sRTNames( (Seat)zBridgeClientIface_get_client(&handle));
122 }
123
124 else if (zBridgeClientIface_israised_rSBoard(&handle))
125 {
126 //Actor has received team names and is ready for start of board.
127 emit sRSBoard( (Seat)zBridgeClientIface_get_client(&handle));
128 }
129
130 else if (zBridgeClientIface_israised_rDealInfo(&handle))
131 {
132 //Actor has received start of board and is ready for deal information.
133 manual = defManual;
134 emit sRDealInfo( (Seat)zBridgeClientIface_get_client(&handle));
135 }
136
137 else if (zBridgeClientIface_israised_rCards(&handle))
138 {
139 //Actor has received deal information and is ready for cards.
140 if (showUser)
141 {
142 //Must show the user the auction info and the table.
143 emit sShowAuction();
144 emit sShowCenter((Team)zBridgeClientIface_get_vulnerability(&handle));
145 }
146
147 emit sRCards((Seat)zBridgeClientIface_get_client(&handle));
148
149 updateCurrentGameInfo = updateGameInfo;
150 }
151
152 else if (zBridgeClientIface_israised_bidInfo(&handle))
153 {
154 //Bidding is finished.
155 if (showUser)
156 {
157 //Must clear bid dialog, clear auction info and show play info.
158 emit sShowBidDialog(false);
159// emit sShowBid((Seat)zBridgeClientIface_get_bidder(&handle), BID_BLANK);
160 emit sShowDummyOnTable((Seat)((zBridgeClientIface_get_declarer(&handle) + 2) & 3));
161 emit sShowPlay();
162 }
163
164 //Set in bid history,
165 bidAndPlay.setBidInfo((Bids)zBridgeClientIface_get_lastBid(&handle),
166 (Bids)zBridgeClientIface_get_bidDouble(&handle),
167 (Seat)zBridgeClientIface_get_bidInfo_value(&handle));
168 }
169
170 else if (zBridgeClientIface_israised_clientPlays(&handle))
171 {
172 //Actor plays a card.
173 if (manual)
174 {
175 //Must get card to play from manual player.
176 Seat player = (Seat)zBridgeClientIface_get_player(&handle);
177 emit sShowYourTurnOnTable(player);
178 emit sEnablePlayer(player);
179 }
180 else
181 {
182 //Must get card to play from automatic play.
183 //Calculate automatic play.
184 QTimer::singleShot(playDelay, this, SLOT(playValue()));
185 }
186 }
187
188 else if (zBridgeClientIface_israised_readyForDummyCards(&handle))
189 {
190 //Actor is ready for dummys cards.
191 emit sReadyForDummyCards((Seat)zBridgeClientIface_get_client(&handle));
192 }
193
194 else if (zBridgeClientIface_israised_getLeader(&handle))
195 {
196 //Get next leader.
197 zBridgeClientIface_raise_newLeader(&handle, bidAndPlay.getNextLeader());
198
199 //State chart run cycle.
200 clientRunCycle();
201 }
202
203 else if (zBridgeClientIface_israised_undoPlay(&handle) || zBridgeClientIface_israised_undoBid(&handle))
204 {
205 //Undo bid.
206 if (showUser)
207 {
208 if (zBridgeClientIface_israised_undoPlay(&handle))
209 {
210 //Rebid when started playing.
211 emit sUndoTrick(0, 0, 0);
212 emit sShowAuction();
213 emit sShowBidDialog(true);
214 }
215
216 //undoBid always follows undoPlay.
217 int undo = zBridgeClientIface_get_undoBid_value(&handle);
218 emit sUndoBid(undo);
219 }
220 }
221
222 else if (zBridgeClientIface_israised_undoTrick(&handle))
223 {
224 //Undo trick.
225 if (showUser)
226 emit sUndoTrick(bidAndPlay.getNoTrick(), bidAndPlay.getNSTricks(),
227 bidAndPlay.getEWTricks());
228 }
229
230 //Can come together with undoBid and must be processed after undoBid.
231 if (zBridgeClientIface_israised_giveBid(&handle))
232 {
233 //Get next bid.
234 if (manual)
235 {
236 //Must get bid from manual player.
237 Seat bidder = (Seat)zBridgeClientIface_get_bidder(&handle);
238 emit sShowYourTurnOnTable(bidder);
239
240 emit sEnableBidder(bidder,
241 (Bids)zBridgeClientIface_get_lastBid(&handle),
242 (Bids)zBridgeClientIface_get_bidEnable(&handle));
243 }
244 else
245 {
246 //Must get bid from automatic player.
247 //Calculate automatic bid.
248 QTimer::singleShot(bidDelay, this, SLOT(bidValue()));
249 }
250 }
251
252 //Can come together with undoBid and must be processed after undoBid.
253 if (zBridgeClientIface_israised_rBid(&handle))
254 {
255 //Actor is ready for bid.
256 emit sRBid((Seat)zBridgeClientIface_get_client(&handle), (Seat)zBridgeClientIface_get_bidder(&handle));
257 }
258
259 //Initially comes together with bidInfo and must be processed after bidInfo.
260 if (zBridgeClientIface_israised_readyForDummy(&handle))
261 {
262 //Actor is ready for dummy to play a card.
263 emit sReadyForDummy((Seat)zBridgeClientIface_get_client(&handle), zBridgeClientIface_get_noTrick(&handle));
264 }
265
266 //Initially comes together with bidInfo and must be processed after bidInfo.
267 if (zBridgeClientIface_israised_readyForPlayer(&handle))
268 {
269 //Actor is ready for player to play a card.
270 emit sReadyForPlayer((Seat)zBridgeClientIface_get_client(&handle),
271 (Seat)zBridgeClientIface_get_player(&handle), zBridgeClientIface_get_noTrick(&handle));
272 }
273
274 //Can Come after undoTrick.
275 if (zBridgeClientIface_israised_synchronize(&handle))
276 {
277 //Synchronization of server and clients.
278 zBridgeClientSync_init(&syncHandle);
279 int syncState = zBridgeClientIface_get_syncState(&handle);
280 zBridgeClientSyncIface_set_syncState(&syncHandle, syncState);
281 synchronizing = true;
282 zBridgeClientSync_enter(&syncHandle);
283 clientSyncActions();
284 }
285}

287void CActorLocal::clientSyncActions()
288{
289 //React to sync client out events.
290 if (zBridgeClientSyncIface_israised_sendAttemptSync(&syncHandle))
291 {
292 //Update Table Manager game info.
293 int syncState = zBridgeClientSyncIface_get_syncState(&syncHandle);
294 if (updateCurrentGameInfo && (syncState == SS)) //Sync Start of Board.
295 {
296 emit sUpdateGame();
297 updateCurrentGameInfo = false;
298 }
299
300 Seat seat = (Seat)zBridgeClientIface_get_client(&handle);
301 emit sAttemptSyncFromClientToServer(seat);
302 }
303
304 else if (zBridgeClientSyncIface_israised_sendConfirmSync(&syncHandle))
305 {
306 //Might pause here (show button etc.).
307 int syncState = zBridgeClientSyncIface_get_syncState(&syncHandle);
308 //Synchronization after bid and before play?
309 if (syncState == SP) //Sync Play.
310 {
311 Seat declarer = bidAndPlay.getBidHistory().getDeclarer();
312 Seat dummy = (Seat)((declarer + 2) % 4);
313 Seat leader = (Seat)((declarer + 1) % 4);
314 zBridgeClientIface_set_declarer(&handle, declarer);
315 zBridgeClientIface_set_dummy(&handle, dummy);
316 zBridgeClientIface_set_leader(&handle, leader);
317 zBridgeClientIface_set_player(&handle, leader);
318 }
319 if (updateGameInfo && (syncState == SS)) //Sync Start of Board.
320 emit sUpdateGameToNextDeal();
321 if (manual && ((syncState == SA) || (syncState == SP) || (syncState == SS) || (syncState == SL)))
322 emit sEnableContinueSync(syncState);
323 else
324 continueSync(); //Never when showUser is true.
325 }
326
327 else if (zBridgeClientSyncIface_israised_okSync(&syncHandle))
328 {
329 synchronizing = false;
330 zBridgeClientIface_raise_allSync(&handle);
331 clientRunCycle();
332 }
333}

466void CActorLocal::continueSync()
467{
468 int syncState = zBridgeClientSyncIface_get_syncState(&syncHandle);
469 if (manual && ((syncState == SA) || (syncState == SP) || (syncState == SS) || (syncState == SL)))
470 emit sDisableContinueSync(zBridgeClientSyncIface_get_syncState(&syncHandle));
471
472 if (showUser &&(syncState == SL))
473 {
474 //Must prepare play view for next trick.
475 emit sClearCardsOnTable();
476
477 //Show number of tricks.
478 emit sShowTricks(bidAndPlay.getEWTricks(), bidAndPlay.getNSTricks());
479 }
480
481 Seat seat = (Seat)zBridgeClientIface_get_client(&handle);
482 emit sConfirmSyncFromClientToServer(seat);
483}

650void CActorLocal::undoBid(bool reBid)
651{
652 int undo;
653
654 if (reBid)
655 {
656 bidAndPlay.resetPlayHistory(); //Could come from play part.
657 bidAndPlay.resetBidHistory();
658 undo = REBID;
659 }
660 else
661 {
662 Bids bid;
663 undo = bidAndPlay.bidUndo(&bid);
664 if (undo >= 0)
665 zBridgeClientIface_set_bidVal(&handle, bid);
666 }
667 zBridgeClientIface_raise_undo(&handle, undo);
668 clientRunCycle();
