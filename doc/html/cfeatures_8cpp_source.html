163 {

165 total += king;
166 setCountCard((Suit)i, KING, king);

168 setCountCard(ANY, KING, total);

170 //Count of queens.
171 total = 0;
172 for (int i = 0; i < 4; i++)
173 {
174 int queen = calcCount((Suit)i, QUEEN, QUEEN, cards);
175 total += queen;
176 setCountCard((Suit)i, QUEEN, queen);
177 }
178 setCountCard(ANY, QUEEN, total);
179
180 //Count of jacks.
181 total = 0;
182 for (int i = 0; i < 4; i++)
183 {
184 int jack = calcCount((Suit)i, JACK, JACK, cards);
185 total += jack;
186 setCountCard((Suit)i, JACK, jack);
187 }
188 setCountCard(ANY, JACK, total);
189
190 //Count of tens.
191 total = 0;
192 for (int i = 0; i < 4; i++)
193 {
194 int ten = calcCount((Suit)i, TEN, TEN, cards);
195 total += ten;
196 setCountCard((Suit)i, TEN, ten);
197 }
198 setCountCard(ANY, TEN, total);
199
200 //Count of aces + kings.
201 for (int i = 0; i < 4; i++)
202 {
203 int rkcb = calcRkcb((Suit)i, cards);
204 setRkcb((Suit)i, rkcb);
205 }
206
207 //How balanced?
208 {
209 int bal = calcBal(count);
210 setBal(bal);
211 }
212
213 //Distribution points.
214 {
215 int dp[4];
216 int dpt = 0;
217
218 for (int suit = 0; suit < 4; suit++)
219 {
220 dp[suit] = (count[suit] <= 2) ? (3 - count[suit]) : 0;
221 dpt += dp[suit];
222 }
223 setDp(NOTRUMP, dpt);
224
225 for (int trump = 0; trump < 4; trump++)
226 {
227 dpt = 0;
228 for (int suit = 0; suit < 4; suit++)
229 {
230 if (suit != trump)
231 dpt += dp[suit];
232 }
233 setDp((Suit)trump, dpt);
234 }
235 }
236
237 //Difference in length (S-H, S-D, S-C, H-D, H-C, D-C).
238 {
239 for (int suit1 = 3; suit1 > 0; suit1--)
240 for (int suit2 = suit1 - 1; suit2 >= 0; suit2--)
241 setDif2((Suit)suit1, (Suit)suit2, count[suit1] - count[suit2]);
242 }
243
244 //Number of suits longer than 3.
245 {
246 int suits = 0;
247 for (int i = 0; i < 4; i++)
248 if (count[i] > 3)
249 suits++;
250 setSuits(suits);
251 }
252
253 //Length of suits.
254 {
255 for (int suit = 0; suit < 4; suit++)
256 setSuitLen((Suit)suit, count[suit]);
257 }
258
259 //Total length of two suits (S+H, S+D, S+C, H+D, H+C, D+C).
260 {
261 for (int suit1 = 3; suit1 > 0; suit1--)
262 for (int suit2 = suit1 - 1; suit2 >= 0; suit2--)
263 setSuitLen2((Suit)suit1, (Suit)suit2, count[suit1] + count[suit2]);
264 }
265 //Length of longer major.
266 {
267 setLMaj((count[SPADES] > count[HEARTS]) ? count[SPADES] : count[HEARTS]);
268 }
269
270 //Length of longer minor.
271 {
272 setLMin((count[DIAMONDS] > count[CLUBS]) ? count[DIAMONDS] : count[CLUBS]);
273 }
274
275 //Is the suit the longest in the hand.
276 {
277 int longest = 0;
278 for (int suit =0; suit < 4; suit++)
279 if (count[suit] > longest)
280 longest = count[suit];
281 for (int suit = 0; suit < 4; suit++)
282 setLongest((Suit)suit , (longest == count[suit]) ? 1 : 0);
283 }
284
285 //Is the suit not shortest in the hand.
286 {
287 int shortest = 20;
288 for (int suit =0; suit < 4; suit++)
289 if (count[suit] < shortest)
290 shortest = count[suit];
291 for (int suit = 0; suit < 4; suit++)
292 setLong((Suit)suit, (shortest != count[suit]) ? 1 : 0);
293 }
294
295 //Weighed count of intermediate cards.
296 {
297 int weighedCount[4];
298
299 for (int i= 0; i < 4; i++)
300 weighedCount[i] = 0;
301
302 for (int i = 0; i < 13; i++)
303 {
304 Suit suit = CARD_SUIT(cards[i]);
305 int face = CARD_FACE(cards[i]);
306 int w = (face == TEN) ? (4) : (face == (TEN-1)) ? (2) : (face == (TEN-2)) ? (1) : (0);
307 weighedCount[suit] += w;
308 }
309 for (int suit = 0; suit < 4; suit++)
310 setInter((Suit)suit, weighedCount[suit]);
311 setInter(ANY, weighedCount[0] + weighedCount[1] + weighedCount[2] + weighedCount[3]);
312 }
313
314 //Loosers.
315 {
316 int tLoosers = 0;
317
318 for (int suit = 0; suit < 4; suit++)
319 {
320 int loosers = count[suit];
321 if (loosers > 3)
322 loosers = 3;
323 int offset = 3 - loosers;
324 for (int j = 0; j < 13; j++)
325 if ((CARD_SUIT(cards[j]) == suit) &&
326 (CARD_FACE(cards[j]) >= (QUEEN + offset)))
327 loosers--;
328 tLoosers += loosers;
329 setLoosers((Suit)suit, loosers);
330 }
331 setLoosers(ANY, tLoosers);
332 }
333
334 //Quick tricks.
335 {
336 int tQuickTricks = 0;
337
338 for (int suit = 0; suit < 4; suit++)
339 {
340 bool ace = false;
341 bool king = false;
342 bool queen = false;
343 for (int i = 0; i < 13; i++)
344 if (CARD_SUIT(cards[i]) == suit)
345 {
346 if (CARD_FACE(cards[i]) == ACE)
347 ace = true;
348 else if (CARD_FACE(cards[i]) == KING)
349 king = true;
350 else if (CARD_FACE(cards[i]) == QUEEN)
351 queen = true;
352 }
353 int quickTricks;
354 if (ace && king)
355 quickTricks = 4;
356 else if (ace && queen)
357 quickTricks = 3;
358 else if (ace)
359 quickTricks = 2;
360 else if (king && queen)
361 quickTricks = 2;
362 else if (king && (count[suit] > 1))
363 quickTricks = 1;
364 else
365 quickTricks = 0;
366
367 tQuickTricks += quickTricks;
368 setQuickTricks((Suit)suit, quickTricks);
369 }
370 setQuickTricks(ANY, tQuickTricks);
371 }
372
373 //Playing tricks.
374 {
375 //High card playing tricks.
376 int hcpTricks = 0;
377 for (int suit = 0; suit < 4; suit++)
378 {
379 bool ace = false;
380 bool king = false;
381 bool queen = false;
382 bool jack = false;
383 bool ten = false;
384 for (int i = 0; i < 13; i++)
385 if (CARD_SUIT(cards[i]) == suit)
386 {
387 if (CARD_FACE(cards[i]) == ACE)
388 ace = true;
389 else if (CARD_FACE(cards[i]) == KING)
390 king = true;
391 else if (CARD_FACE(cards[i]) == QUEEN)
392 queen = true;
393 else if (CARD_FACE(cards[i]) == JACK)
394 jack = true;
395 else if (CARD_FACE(cards[i]) == TEN)
396 ten = true;
397 }
398 if (ace && king && queen)
399 hcpTricks += 6;
400 else if ((ace && king && jack) || (ace && queen && jack))
401 hcpTricks += 5;
402 else if ((ace && king) || (ace && queen && ten) || (ace && jack && ten) || (king && queen && jack))
403 hcpTricks += 4;
404 else if ((ace && queen) || (king && queen && ten) || (king && jack && ten))
405 hcpTricks += 3;
406 else if (ace || (king && queen) || (king && jack && (count[suit] > 2)) || (queen && jack && ten))
407 hcpTricks += 2;
408 else if (((king && (count[suit] > 1)) || (queen && jack && (count[suit] > 2))))
409 hcpTricks += 1;
410 else
411 hcpTricks += 0;
412 }
413 //Long suit playing tricks.
414 for (int suit = 0; suit < 4; suit++)
415 {
416 //One trick for each card over 3 in the trump suit.
417 int lsptTricks = 0;
418 if (count[suit] > 3)
419 lsptTricks += (count[suit] - 3) * 2;
420
421 //Half a trick for the 4. card and one trick for the 5. and
422 // subsequent cards in side suits.
423 int lspTricks = 0;
424 for (int i = 0; i < 4; i++)
425 if (i != suit)
426 {
427 if (count[i] >= 4)
428 {
429 lspTricks += 1; //Half a trick for the 4. card in a side suit.
430 for (int j = 5; j <= count[i]; j++)
431 lspTricks += 2; //One trick for the 5. and subsequent cards in a side suit.
432 }
433 }
434 setPlayingTricks((Suit)suit, hcpTricks + lspTricks + lsptTricks);
435 if (lsptTricks == 0)
436 setPlayingTricks(NOTRUMP, hcpTricks + lspTricks);
437 }
438 }
