65void CGamesDoc::readGames(QTextStream &original, QTextStream &played,
66 QString &event, ScoringMethod scoringMethod) throw(PlayException)
67{
68 ScoringMethod originalScoringMethod = NOSCORE;
69
70 //Clean up for new game.
71 clearGames(NOSCORE);
72
73 this->event = event;
74
75 //Original games (check that there are original games).
76 if (original.device() != 0)
77 {
78 readGames(original, event, true);
79 if (games.size() > 0)
80 {
81 dealType = ORIGINAL_DEAL;
82 if ((this->scoringMethod == RUBBER) || (this->scoringMethod == PRACTICE))
83 this->scoringMethod = NOSCORE;
84 originalScoringMethod = this->scoringMethod;
85 }
86 }
87
88 //Played games (check that there are played games).
89 if (played.device() != 0)
90 {
91 readGames(played, event, false);
92 }
93
94 if (this->scoringMethod == NOSCORE)
95 {
96 this->scoringMethod = scoringMethod;
97 }
98
99 if ((dealType != ORIGINAL_DEAL) || ((dealType == ORIGINAL_DEAL) && originalScoringMethod == NOSCORE))
100 computerPlays =((this->scoringMethod == MP) || (this->scoringMethod == IMP));
101}

520void CGamesDoc::setResult(GameType gameType, CBidHistory &bidHistory, CPlayHistory &playHistory, QString &westName,

566void CGamesDoc::determineEvents(QTextStream &original, QStringList &events)
567{
568 QString line;
569 QString event;
570
571 while (!original.atEnd())
572 {
573 line = original.readLine();
574
575 if (line.indexOf("[EVENT", 0, Qt::CaseInsensitive) == 0)
576 {
577 int firstInx = line.indexOf('"');
578 int lastInx = line.lastIndexOf('"');
579
580 //Check for syntax (silently discard in case of syntax error).
581 if ((firstInx == -1) || (firstInx == lastInx))
582 continue;
583
584 //Handle header (##) and substitution (#).
585 if (line.indexOf("##", firstInx + 1) == (firstInx + 1))
586 event = line.mid(firstInx + 3, lastInx - (firstInx + 3));
587 else if (line.indexOf('#', firstInx + 1) != (firstInx + 1))
588 event = line.mid(firstInx + 1, lastInx - (firstInx + 1));
589 else
590 continue;
591
592 if (event == "?")
593 event = "";
594
595 if (events.indexOf(event) == -1)
596 events.append(event);
597 }
598 }

748int CGamesDoc::getDuplicateScore(int gameIndex, int auctionAndPlayIndex, bool ns)
749{
750 assert(gameIndex < games.size());
751 assert(auctionAndPlayIndex < games[gameIndex]->auctionAndPlay.size());
752
753 Seat dealer, declarer;
754 Team vulnerable;
755 Bids contract, contractModifier;
756 int result;
757 int duplicateScore;
758
759 dealer = games[gameIndex]->dealer;
760 vulnerable = games[gameIndex]->vulnerable;
761 declarer = games[gameIndex]->auctionAndPlay[auctionAndPlayIndex]->declarer;
762 contract = games[gameIndex]->auctionAndPlay[auctionAndPlayIndex]->contract;
763 contractModifier = games[gameIndex]->auctionAndPlay[auctionAndPlayIndex]->contractModifier;
764 result = games[gameIndex]->auctionAndPlay[auctionAndPlayIndex]->result;
765
766 //Was it a pass game?
767 if (contract == BID_PASS)
768 return 0;
769
770 Suit suit = BID_SUIT(contract);
771 int level = BID_LEVEL(contract);
772
773 bool declarerVulnerable = (vulnerable == BOTH) ||
774 ((vulnerable == NORTH_SOUTH) && ((declarer == SOUTH_SEAT) || (declarer == NORTH_SEAT))) ||
775 ((vulnerable ==EAST_WEST) && ((declarer == EAST_SEAT) || (declarer == WEST_SEAT)));
776
777 //Contract made?
778 if (result >= (level + 6))
779 {
780 //Contract points.
781 int contractPoints = ISNOTRUMP(suit) ? (40) : ISMAJOR(suit) ? (30) : (20);
782 if (level > 1)
783 contractPoints += (ISMINOR(suit) ? (20) : (30)) * (level - 1);
784 if (IS_DOUBLE_BID(contractModifier))
785 contractPoints *=2;
786 else if (IS_REDOUBLE_BID(contractModifier))
787 contractPoints *=4;
788
789 //Overtrick points.
790 int prTrick;
791 if (IS_DOUBLE_BID(contractModifier))
792 prTrick = declarerVulnerable ? (200) : (100);
793 else if (IS_REDOUBLE_BID(contractModifier))
794 prTrick = declarerVulnerable ? (400) : (200);
795 else
796 prTrick = ISMINOR(suit) ? (20) : (30);
797 int overtrickPoints = prTrick * (result - (level + 6));
798
799 //Slam bonus.
800 int slamBonus;
801 slamBonus = (level >= 6) ? (declarerVulnerable ? (750) : (500)) : (0);
802 if (level == 7)
803 slamBonus *= 2;
804
805 //Doubled or redoubled bonus.
806 int doubleBonus = IS_DOUBLE_BID(contractModifier) ? (50) : IS_REDOUBLE_BID(contractModifier) ? (100) :(0);
807
808 //Game or part game bonus.
809 int gameBonus = (contractPoints < 100) ? (50) : declarerVulnerable ? (500) : (300);
810
811 duplicateScore = contractPoints + overtrickPoints + slamBonus + doubleBonus + gameBonus;
812 }
813
814 //Contract not made.
815 else
816 {
817 int underTrick_1, underTrick_2_3, underTrick_4_;
818
819 if (declarerVulnerable)
820 {
821 if (IS_DOUBLE_BID(contractModifier))
822 {
823 underTrick_1 = 200;
824 underTrick_2_3 = 300;
825 underTrick_4_ = 300;
826 }
827 else if (IS_REDOUBLE_BID(contractModifier))
828 {
829 underTrick_1 = 400;
830 underTrick_2_3 = 600;
831 underTrick_4_ = 600;
832 }
833 else
834 underTrick_1 = underTrick_2_3 = underTrick_4_ = 100;
835 }

880float CGamesDoc::getDuplicatePointBoard(int gameIndex, int auctionAndPlayIndex,
881 int scoringMethod, bool ns)
882{
883 float point = 0;
884
885 assert(gameIndex < games.size());
886
887 int maxNumberAuctionAndPlay = getMaxNumberAuctionAndPlay();
888 assert(maxNumberAuctionAndPlay >= 2);
889
890 //Has the game been played?
891 if (auctionAndPlayIndex == -1)
892 return (scoringMethod == MP) ? (maxNumberAuctionAndPlay - 1) : (0);
893
894 assert(auctionAndPlayIndex < games[gameIndex]->auctionAndPlay.size());
895
896 int nrPlayed = games[gameIndex]->auctionAndPlay.size();
897 if (nrPlayed == 1)
898 return (scoringMethod == MP) ? (maxNumberAuctionAndPlay - 1) : (0);
899
900 Seat declarer = games[gameIndex]->auctionAndPlay[auctionAndPlayIndex]->declarer;
901
902 //Allocate score table and get duplicate scores.
903 int *scores = new int[nrPlayed];
904 for (int i = 0; i < nrPlayed; i++)
905 scores[i] = getDuplicateScore(gameIndex, i, true);
906
907 if (scoringMethod == MP)
908 {
909 //Calculate point based on MP.
910 int *gt = new int[nrPlayed];
911 int *eq = new int[nrPlayed];
912
913 for (int i = 0; i < nrPlayed; i++)
914 gt[i] = eq[i] = 0;
915
916 for (int i = 0; i < nrPlayed; i++)
917 for (int j = 0; j < nrPlayed; j++)
918 {
919 if (scores[i] < scores[j])
920 gt[i]++;
921 else if (scores[i] == scores[j])
922 eq[i]++;
923 }
924
925 point = 2*nrPlayed - 2 * (gt[auctionAndPlayIndex] + 1);
926 for (int j = 1; j < eq[auctionAndPlayIndex]; j++)
927 point += 2*nrPlayed - 2 * (gt[auctionAndPlayIndex] + 1 + j);
928 point = point / eq[auctionAndPlayIndex];
929 if (!ns && ((declarer == WEST_SEAT) || (declarer == EAST_SEAT)))
930 point = 2*nrPlayed - 2 - point;
931
932 delete []eq;
933 delete []gt;
934
935 //Correct for non played games.
936 if (maxNumberAuctionAndPlay > nrPlayed)
937 point += (maxNumberAuctionAndPlay - nrPlayed);
938
939 //Normalize to %.
940 point = point * 100 / (2 * maxNumberAuctionAndPlay - 2);
941 }
942 else
943 {
944 //Calculate point based on IMP. First played games.
945 for (int i = 0; i < nrPlayed; i++)
946 if (auctionAndPlayIndex != i)
947 {
948 int diff = scores[auctionAndPlayIndex] - scores[i];
949 int k = -1;
950 while (abs(diff) > IMP_TABLE[++k]) ;
951 point += (diff > 0) ? k : -k;
952 }
953 //Next non played games (IMP is 0 for non played games).
954 for (int i = nrPlayed; i < maxNumberAuctionAndPlay; i++)
955 {
956 int diff = scores[auctionAndPlayIndex] - 0;
957 int k = -1;
958 while (abs(diff) > IMP_TABLE[++k]) ;
959 point += (diff > 0) ? k : -k;
960
961 }
962 if (!ns &&((declarer == WEST_SEAT) || (declarer == EAST_SEAT)))
963 point = -point;
964
965 //Normalize.
966 point /= (maxNumberAuctionAndPlay - 1);
967 }
968 delete []scores;
969
970 return point;
971}
