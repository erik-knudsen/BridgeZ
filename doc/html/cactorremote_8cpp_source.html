52void CActorRemote::xmitPBNFiles(QTextStream &originalStream, QTextStream &playedStream,
53 ScoringMethod scoringMethod)
54{
55 originalStream.seek(0);
56 playedStream.seek(0);
57
58 //First transmit original data.
59 COriginalPBNStartMsg originalPBNStartMsg(scoringMethod);
60 emit sendLine(originalPBNStartMsg.line);
61
62 while (!originalStream.atEnd())
63 emit sendLine(originalStream.readLine() + "\r\n");
64
65 CEscapePBNMsg escapePBNMsg;
66 emit sendLine(escapePBNMsg.line);
67
68 //Next transmit played data.
69 CPlayedPBNStartMsg playedPBNStartMsg;
70 emit sendLine(playedPBNStartMsg.line);
71
72 while (!playedStream.atEnd())
73 emit sendLine(playedStream.readLine() + "\r\n");
74
75 emit sendLine(escapePBNMsg.line);
76}

280void CActorRemote::receiveLine(QString line, bool connected)
281{
282 try
283 {
284 //Determine type.
285 MsgType msgType = ::getMessageType(line);
286 switch (msgType)
287 {
288 case CONNECT_MSG:
289 {
290 //A connect message has been received
291 CConnectMsg connectMsg(line);
292
293 //If already connected then the connect message should be discarded.
294 //The message will be received when a new session is requested from the server. In this
295 //case the client is not restarted completely from scratch, but the already established
296 //connection is kept. So the connect message is not needed.
297 if (!connected)
298 emit sConnect(connectMsg.name, connectMsg.seat, connectMsg.protocol);
299 break;
300 }
301
302 case RTNAMES_MSG:
303 {
304 //A redy for team names message has been received.
305 CRTNamesMsg rTNames(line);
306 emit sRTNames(rTNames.seat);
307 break;
308 }
309
310 case RSBOARD_MSG:
311 {
312 //A ready for start of board message has been received.
313 CRSBoardMsg rSBoard(line);
314 emit sRSBoard(rSBoard.seat);
315 break;
316 }
317
318 case RDEALINFO_MSG:
319 {
320 //A ready for deal info message has been received.
321 CRDealInfoMsg rDealInfoMsg(line);
322 emit sRDealInfo(rDealInfoMsg.seat);
323 break;
324 }
325
326 case RCARDS_MSG:
327 {
328 //A ready for cards message has been received.
329 CRCardsMsg rCardsMsg(line);
330 emit sRCards(rCardsMsg.seat);
331 break;
332 }
333
334 case BID_MSG:
335 {
336 //A bid message has been received.
337 CBidMsg bidMsg(line);
338 emit sBid(bidMsg.bidder, bidMsg.bid);
339 break;
340 }
341
342 case RBID_MSG:
343 {
344 //A ready for bid message has been received.
345 CRBidMsg rBidMsg(line);
346 emit sRBid(rBidMsg.player, rBidMsg.bidder);
347 break;
348 }
349
350 case PLAYERPLAYS_MSG:
351 {
352 //A player plays message has been received.
353 CPlayerPlaysMsg playerPlaysMsg(line);
354 emit sPlayerPlays(playerPlaysMsg.player, playerPlaysMsg.card);
355 break;
356 }
357
358 case READYFORPLAYER_MSG:
359 {
360 //A ready for player message has been received.
361 CReadyForPlayerMsg readyForPlayerMsg(line);
362 emit sReadyForPlayer(readyForPlayerMsg.seat, readyForPlayerMsg.player, readyForPlayerMsg.trick);
363 break;
364 }
365
366 case READYFORDUMMY_MSG:
367 {
368 //A ready for dummy message has been received.
369 CReadyForDummyMsg readyForDummyMsg(line);
370 emit sReadyForDummy(readyForDummyMsg.seat, readyForDummyMsg.trick);
371 break;
372 }
373
374 case READYFORDUMMYCARDS_MSG:
375 {
376 //A ready for dummys cards message has been received.
377 CReadyForDummyCardsMsg readyForDummyCardsMsg(line);
378 emit sReadyForDummyCards(readyForDummyCardsMsg.seat);
379 break;
380 }
381
382 case ATTEMPT_SYNCHRONIZE_MSG:
383 {
384 //An attempt synchronize message has been received.
385 CAttemptSynchronizeMsg attemptSynchronize(line);
386 emit sAttemptSyncFromClientToServer(attemptSynchronize.seat);
387 break;
388 }
389
390 case CONFIRM_SYNCHRONIZE_MSG:
391 {
392 //A confirm synchronize message has been received.
393 CConfirmSynchronizeMsg confirmSynchronize(line);
394 emit sConfirmSyncFromClientToServer(confirmSynchronize.seat);
395 break;
396 }
397
398 default:
399 throw NetProtocolException("Net - Illegal message: " + line.toStdString());
400 break;
401
402 }
403 }
404 catch (NetProtocolException &e)
405 {
406 //There was an error in the message.
407 QMessageBox::critical(0, tr("ZBridge"), e.what());
408 }
