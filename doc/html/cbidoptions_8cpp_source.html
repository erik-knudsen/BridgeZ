38CBidOptions::CBidOptions(CZBridgeApp *app, CZBridgeDoc *doc, QWidget *parent) :
39 QDialog(parent),

42 //Set up user interface.

72 ui->eastWestConvention->setCurrentIndex(ewIndex);

126void CBidOptions::on_newConvention_clicked()
127{
128 int i;
129 QString refConventionName;
130
131 //Determine reference bid option set.
132 if ((nsIndex >= 0) && (ui->northSouth->isChecked()))
133 refConventionName = ui->northSouthConvention->currentText();
134 else
135 if ((ewIndex >= 0) && (ui->eastWest->isChecked()))
136 refConventionName = ui->eastWestConvention->currentText();
137
138 //Activate new convention dialog (to determine name of new bid option set).
139 CNewConvention newConventionDialog(refConventionName, this);
140 if (newConventionDialog.exec() == QDialog::Accepted)
141 {
142 //Name is accepted.
143 QString newConventionName = newConventionDialog.getNewConventionName();
144
145 //Check name of new bid option set.
146 if (newConventionName.isEmpty())
147 {
148 QMessageBox::warning(this, tr("ZBridge"),
149 tr("Illegal name of new bid option set.\nThe name must not be empty."));
150 return;
151 }
152
153 for (i = 0; i < bidOptions->size(); i++)
154 if ((*bidOptions)[i].name == newConventionName)
155 {
156 QMessageBox::warning(this, tr("ZBridge"),
157 tr("Illegal name of new bid option set.\nThe bid option set already exists."));
158 return;
159 }
160 for (i = 0; i < bidOptions->size(); i++)
161 if ((*bidOptions)[i].name == refConventionName)
162 break;
163
164 //Name is ok. Activate bid options property sheet to determine bid options for the new
165 //bid option set.
166 CBidOptionDoc bidOption;
167 bidOption = doc->getDefaultBidOption();
168 if (i < bidOptions->size())
169 bidOption = (*bidOptions)[i];
170 CBidOptionsPropSheet biddingOptsDialog(bidOption, app, doc, this);
171 if (biddingOptsDialog.exec() == QDialog::Accepted)
172 {
173 //The new bid option set was accepted. Add it to available bid option sets.
174 bidOption = biddingOptsDialog.getBidOption();
175 bidOption.name = newConventionName;
176 bidOptions->append(bidOption);
177 ui->northSouthConvention->addItem(bidOption.name);
178 ui->eastWestConvention->addItem(bidOption.name);
179 ui->DeleteConvention->setEnabled(true);
180 ui->editConvention->setEnabled(true);
181
182 doc->SaveBidOptions();
183 }
184 }
185}

190void CBidOptions::on_editConvention_clicked()
191{
192 QString curConventionName;
193 int i;
194
195 //Determine which bid option set to edit.
196 if (ui->northSouth->isChecked())
197 curConventionName = ui->northSouthConvention->currentText();
198 else
199 curConventionName = ui->eastWestConvention->currentText();
200
201 for (i = 0; i < bidOptions->size(); i++)
202 if ((*bidOptions)[i].name == curConventionName)
203 break;
204
205 //Activate bid option property sheet to edit the current bid option set.
206 CBidOptionsPropSheet biddingOptsDialog((*bidOptions)[i], app, doc, this);
207 if (biddingOptsDialog.exec() == QDialog::Accepted)
208 {
209 //Editing was accepted. Save the changes.
210 (*bidOptions)[i] = biddingOptsDialog.getBidOption();
211
212 doc->SaveBidOptions();
213 }
214}

219void CBidOptions::on_DeleteConvention_clicked()
220{
221 int delIndex;
222 QString delConventionName;
223
224 //Determine which bid option set to delete.
225 if (ui->northSouth->isChecked())
226 {
227 delConventionName = ui->northSouthConvention->currentText();
228 delIndex = nsIndex;
229 }
230 else
231 {
232 delConventionName = ui->eastWestConvention->currentText();
233 delIndex = ewIndex;
234 }
235
236 int i;
237 for (i = 0; i < bidOptions->size(); i++)
238 if ((*bidOptions)[i].name == delConventionName)
239 break;
240
241 //Delete the bid option set.
242 bidOptions->removeAt(i);
243 ui->northSouthConvention->removeItem(delIndex);
244 ui->eastWestConvention->removeItem(delIndex);
245 if (bidOptions->size() == 0)
246 {
247 ui->DeleteConvention->setEnabled(false);
248 ui->editConvention->setEnabled(false);
249 }
250
251 //Save the changes.
252 doc->SaveBidOptions();
253}
